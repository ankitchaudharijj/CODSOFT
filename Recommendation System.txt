class Book:
    def __init__(self, title, genre, author):
        self.title = title
        self.genre = genre
        self.author = author

    def __str__(self):
        return f"Title: {self.title}, Genre: {self.genre}, Author: {self.author}"


class RecommendationSystem:
    def __init__(self):
        self.books = [
            Book("The Great Gatsby", "Classic", "F. Scott Fitzgerald"),
            Book("To Kill a Mockingbird", "Fiction", "Harper Lee"),
            Book("1984", "Dystopian", "George Orwell"),
            Book("The Catcher in the Rye", "Fiction", "J.D. Salinger"),
            Book("Moby Dick", "Adventure", "Herman Melville"),
            Book("The Hobbit", "Fantasy", "J.R.R. Tolkien"),
            Book("The Diary of a Young Girl", "Biography", "Anne Frank")
        ]

    def recommend_books(self, genre_preference):
        print(f"Recommended books based on your preference for {genre_preference} genre:")
        for book in self.books:
            if book.genre.lower() == genre_preference.lower():
                print(book)


def main():
    recommender = RecommendationSystem()

    print("Welcome to the Book Recommendation System!")
    genre_preference = input("Enter a genre you're interested in (e.g., Fiction, Classic, Fantasy): ")

    recommender.recommend_books(genre_preference)


if __name__ == "__main__":
    main()
